// AUTO-GUIDE/scanner.js
const fs = require('fs');
const path = require('path');

class ProjectScanner {
  scan() {
    console.log('ğŸ” Starting automated scan...');
    
    const issues = [];
    
    // Check for N+1 queries
    if (this.fileExists('server/auth-middleware.ts')) {
      const content = fs.readFileSync('server/auth-middleware.ts', 'utf8');
      if (content.includes('await supabase.from("users")')) {
        issues.push({
          type: 'security',
          severity: 'high',
          title: 'N+1 Query in Auth Middleware',
          fix_id: 'HP001'
        });
      }
    }
    
    // Check for email injection
    if (this.fileExists('server/email.ts')) {
      const content = fs.readFileSync('server/email.ts', 'utf8');
      if (content.includes('${data.') && !content.includes('escape(')) {
        issues.push({
          type: 'security',
          severity: 'high',
          title: 'Email HTML Injection Risk',
          fix_id: 'HP003'
        });
      }
    }
    
    // Update state
    this.updateState(issues);
    console.log(`âœ… Found ${issues.length} issues`);
  }
  
  fileExists(filePath) {
    return fs.existsSync(filePath);
  }
  
  updateState(issues) {
    const statePath = './PROJECT_STATE.json';
    const state = JSON.parse(fs.readFileSync(statePath, 'utf8'));
    
    state.detected_issues = {
      security: issues.filter(i => i.type === 'security'),
      performance: issues.filter(i => i.type === 'performance'),
      ux: issues.filter(i => i.type === 'ux')
    };
    
    state.last_scan = new Date().toISOString();
    
    fs.writeFileSync(statePath, JSON.stringify(state, null, 2));
  }
}

// Export for use
if (typeof module !== 'undefined') {
  module.exports = ProjectScanner;
}
